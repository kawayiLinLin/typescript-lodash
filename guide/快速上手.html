# 快速上手

# 安装

npm install -S typescript-lodash

# 使用

import { string } from "typescript-lodash"

type IString = "今天很高兴"

type TypeExample = string.Split<IString, "">

// 得到
type TypeResult = ["今", "天", "很", "高", "兴"]

详细使用请参考：文档

注意

请确保你项目的 TypeScript 版本 >= 4.5.0。
